movement_religious_majority = {
	category = movement_category_religious
	
	ideology = ideology_traditionalist_movement
	character_ideologies = {
		ideology_theocrat
		ideology_traditionalist
		ideology_turkfm_islamist #GOTB
		ideology_usfp_social_gospel_leader #USFP
	}

	can_cause_obstinance = yes
	
	creation_trigger = {
		NOT = {
			has_law_or_variant = law_type:law_state_atheism
		}
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {}

	disband_trigger = {
		scope:political_movement ?= {
			religion = {
				NOT = { is_state_religion = root }
			}		
		}
	}
	
	religion_selection_trigger = {
		is_state_religion = scope:country
	}
	religion_selection_weight = {
		value = 100
	}
	
	character_support_trigger = {
		religion = scope:religion
		OR = {
			is_interest_group_type = ig_devout
			character_is_pro_religious_majority = yes #Compatibility Trigger
		}		
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				scope:interest_group ?= {
					is_interest_group_type = ig_devout
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_theocrat
					has_ideology = ideology:ideology_turkfm_islamist #GOTB
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_integralist
			}
			multiply = {
				value = 2
			}
		}
	}	
	
	pop_support_trigger = {
		religion = scope:religion
	}

    pop_support_factors = {
        movement_support_low_urbanization
        movement_support_low_literacy
        movement_support_clergymen
        movement_support_aristocrats
        movement_support_peasants
        movement_support_corporatism_tech
    }

    pop_support_weight = {
        if = {
            limit = {
                exists = scope:political_movement
                scope:political_movement ?= {
                    OR = {
                        is_being_suppressed = yes
                        is_being_bolstered = yes
                    }
                }
            }
            if = {
                limit = {
                    scope:political_movement = {
                        is_being_suppressed = yes
                    }
                }
                add = {
                    value = suppression_base_value
                    multiply = {
                        value = -0.5
                        add = owner.modifier:country_suppression_attraction_factor
                    }
                    desc = "concept_suppressed"
                }
            }
            else = {
                add = {
                    value = bolstering_base_value
                    multiply = {
                        value = 0.5
                        add = owner.modifier:country_bolster_attraction_factor
                    }
                    desc = "concept_bolstered"
                }
            }
        }

        add = {
            desc = "URBAN_STATE" # Increases support in rural states, decreases in urban
            value = 0.5
            subtract = state.state_urbanization_rate
            multiply = 10
        }

        add = {
            value = 10
            desc = "POP_BASE_SUPPORT"
        }

        if = {
            limit = {
                is_pop_type = clergymen
            }
            add = {
                value = 45
                desc = "POP_CLERGYMEN"
            }
        }
        else_if = {
            limit = {
                is_pop_type = aristocrats
            }
            add = {
                value = 15
                desc = "POP_ARISTOCRATS"
            }
        }
        else_if = {
            limit = {
                is_pop_type = peasants
            }
            add = {
                value = 10
                desc = "POP_PEASANTS"
            }
        }

		add = { # Weakened by increasing literacy rates
            desc = "POP_LITERACY"
            this = {
                value = literacy_rate
                multiply = -2
            }
        }

		if = { # Discriminated pops are less inclined to support the state church
            limit = {
                pop_acceptance < acceptance_status_5
            }
            multiply = {
                desc = "POP_ACCEPTANCE"
                value = pop_acceptance
                divide = 100
                add = 0.25
            }
        }

        if = {
            limit = {
                owner ?= {
                    has_technology_researched = corporatism
                }
            }
            multiply = {
                value = 1.5
                desc = "CORPORATISM_TECH"
            }
        }

        if = {
            limit = {
                owner ?= {
                    any_political_movement = {
                        is_political_movement_type = movement_legitimist
                    }
                }
            }
            multiply = {
                value = 0.25
            }
        }

		#USFP
		multiply = {
			value = usfp_harlem_renaissance_religious_attraction_mult
			desc = "harlem_renaissance_movement_attraction"
		}
    }

    revolution = {
        weight = {
            value = 100
        }

        state_weight = {
            value = 1

            if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.75
                    }
                }
                add = 75
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.5
                    }
                }
                add = 50
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.25
                    }
                }
                add = 25
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.1
                    }
                }
                add = 10
            }
        }
    }

    secession = {
        possible = {
            political_movement_support >= 0.05
            scope:clout <= 0
        }

        weight = {
            value = 1000
        }

        state_weight = {
            value = 0

            if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.75
                    }
                }
                add = 250
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.5
                    }
                }
                add = 150
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.25
                    }
                }
                add = 50
            }
            else_if = {
                limit = {
                    religion_percent_state = {
                        target = scope:political_movement.religion
                        value >= 0.1
                    }
                }
                add = 10
            }
        }
    }

    additional_radicalism_factors = {
        if = {
            limit = {
                owner.ruler ?= {
                    NOT = { religion = root.religion }
                }
            }

            add = {
                value = 0.1
                desc = "RULER_WRONG_RELIGION"
            }
        }

        add = {
            value = "owner.average_sol"
            subtract = "owner.average_sol_for_religion(scope:religion)"
            multiply = 0.05
            max = 0.5
            min = 0
            desc = "STANDARD_OF_LIVING_FOR_RELIGION"
        }

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								religion ?= ROOT.religion
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_RELIGIOUS_REVOLUTION"
				value = 0.15
			}
		}
    }
}

movement_religious_minority = {
	category = movement_category_religious

	ideology = ideology_traditionalist_minoritarian_movement
	character_ideologies = {
		ideology_traditionalist_minoritarian
		ideology_usfp_universalist #USFP
		ideology_turkfm_ottomanist #GOTB
	}

	can_cause_obstinance = yes

	creation_trigger = {
		NAND = {
			c:BIC ?= this
			has_technology_researched = pan-nationalism
		}
	}
	creation_weight = {
		value = 200
	}
	on_created = {}

    disband_trigger = {
        OR = {
            scope:political_movement ?= {
                religion = {
                    is_state_religion = root
                }
                root = {
                    religion_percent_country = {
                        target = scope:political_movement.religion
                        value <= 0.08
                    }
                }
            }
            AND = {
                c:BIC ?= this
                has_technology_researched = pan-nationalism
            }
        }
    }

	on_disbanded = {
		if = {
			limit = {
				owner ?= {
					c:BIC ?= this
					has_technology_researched = pan-nationalism
					NOT = {
						any_political_movement = {
						    is_political_movement_type = movement_hindu_nationalist
						}
					}
				}
			}
			owner ?= {
				create_political_movement = { type = movement_hindu_nationalist }
			}
		}
		if = {
			limit = {
				owner ?= {
					c:BIC ?= this
					has_technology_researched = pan-nationalism
					NOT = {
						any_political_movement = {
						    is_political_movement_type = movement_muslim_nationalist
						}
					}
				}
			}
			owner ?= {
				create_political_movement = { type = movement_muslim_nationalist }
			}
		}
	}

	religion_selection_trigger = {
		NOT = { is_state_religion = scope:country }
		NOT = { rel:atheist = this }
		scope:country = {
			religion_percent_country = {
				target = root
				value >= 0.1
			}
		}
	}
	religion_selection_weight = {
		value = 100
	}

	character_support_trigger = {
		religion = scope:religion
		OR = {
			AND = {
				owner ?= {
					has_law_or_variant = law_type:law_state_atheism
				}
				character_hates_state_atheism = yes #Compatibility Trigger
			}
			character_is_pro_religious_minority = yes #Compatibility Trigger
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				scope:interest_group ?= {
					is_interest_group_type = ig_devout
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						owner ?= {
							has_law = law_type:law_state_atheism
						}
						character_hates_state_atheism = yes #Compatibility Trigger
					}
					character_is_pro_religious_minority = yes #Compatibility Trigger
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				NOT = {
					scope:interest_group ?= {
						is_interest_group_type = ig_devout
					}
				}
				OR = {
					has_progressive_ideology = yes
					has_socialist_ideology = yes
				}
			}
			multiply = 0.05
		}
	}

	pop_support_trigger = {
		religion = scope:religion
	}

    pop_support_factors = {
        movement_support_poverty
        movement_support_low_literacy
        movement_support_religious_discrimination
        movement_support_clergymen
        movement_support_aristocrats
        movement_support_peasants
        movement_support_corporatism_tech
    }

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			value = 10
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
			}
			add = { # Increases likelihood of support if impoverished
				desc = "POP_POVERTY"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 45
				desc = "POP_CLERGYMEN"
			}
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 10
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 10
				desc = "POP_PEASANTS"
			}
		}

		add = { # Weakened by increasing literacy rates
			desc = "POP_LITERACY"
			this = {
				value = literacy_rate
				multiply = -2
			}
		}

		if = {
			limit = {
				owner ?= {
					OR = {
						c:RUS ?= this
						c:PRU ?= this
						c:AUS ?= this
					}
				}
				culture = cu:polish
			}
			multiply = {
				desc = "POP_POLISH_NATIONALISM"
				value = 0.05
			}
		}
		
		# Grey_PLP
		multiply = {
			value = mog_plp_movement_religious_minority_pop_support_weight_mult
			desc = "STANDARD_OF_LIVING_FOR_RELIGION"								# Desc only shows when value =/= 1
		}
		
		if = {
			limit = {
				state = {
					state_religious_acceptance = {
						target = PREV.religion
						value < religious_acceptance_high
					}
				}
			}
			add = { # From +2 at 80 to +10 at 0
				desc = "POP_RELIGIOUS_DISCRIMINATION"
				value = 100
				subtract = pop_acceptance
				divide = 10
			}
		}
		else = { # Non-discriminated pops are less likely to form separatist movements
			multiply = {
				desc = "POP_ACCEPTANCE"
				value = 0.25
			}
		}

		if = {
			limit = {
				owner ?= {
					has_technology_researched = corporatism
				}
			}
			multiply = {
				value = 1.5
				desc = "CORPORATISM_TECH"
			}
		}

		# Favor cultural movement if necessary
		if = {
			limit = {
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
						this.culture ?= root.culture
					}
				}
			}
			if = {
				limit = {
					owner = { cultural_acceptance_base = { target = root.culture value <= acceptance_status_1 } }	# BugFix from PLP (root.culture, not culture)
				}
				multiply = 0.1
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = root.culture value <= acceptance_status_2 } }	# BugFix from PLP (root.culture, not culture)
				}
				multiply = 0.25
			}
			else_if = {
				limit = {
					owner = { cultural_acceptance_base = { target = root.culture value <= acceptance_status_3 } }	# BugFix from PLP (root.culture, not culture)
				}
				multiply = 0.5
			}
		}

		#USFP
		multiply = {
			value = usfp_harlem_renaissance_religious_attraction_mult
			desc = "harlem_renaissance_movement_attraction"
		}
	}

	secession = {
		possible = {
			owner ?= {
				any_scope_state = {
					religion_percent_state = {
						target = root.religion
						value >= com_movement_religious_minority_secession_possible_religion_percent_state #vanilla value: 0.10
					}
				}
				trigger_if = {
					limit = {
						turkfm_revolt_mechanics_trigger = yes
					}
					turkfm_minority_revolt_possible_trigger = yes
				}
			}
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = 0

			if = {
				limit = {
					is_capital = no
				}

				if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= com_movement_religious_minority_secession_state_weight_religion_percent_state_1 #vanilla value: 0.75
						}
					}
					add = 250
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= com_movement_religious_minority_secession_state_weight_religion_percent_state_2 #vanilla value: 0.5
						}
					}
					add = 150
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= com_movement_religious_minority_secession_state_weight_religion_percent_state_3 #vanilla value: 0.25
						}
					}
					add = 50
				}
				else_if = {
					limit = {
						religion_percent_state = {
							target = scope:political_movement.religion
							value >= com_movement_religious_minority_secession_state_weight_religion_percent_state_4 #vanilla value: 0.1
						}
					}
					add = 10
				}
			}
		}

		target_fraction_of_states = {
			value = political_movement_identity_support
			add = scope:clout
			multiply = 3
			if = {
				limit = {
					grefm_is_active = yes
					owner = {
						turkfm_revolt_mechanics_trigger = yes
					}
				}
				multiply = 0
				add = turkfm_target_fraction_of_states_value
			}
			min = {
				value = com_movement_religious_minority_secession_target_fraction_of_states_min #vanilla value: 0.15
				if = {
					limit = {
						grefm_is_active = yes
						owner = {
							turkfm_revolt_mechanics_trigger = yes
						}
					}
					multiply = 0
					add = turkfm_movement_religious_minority_secession_target_fraction_of_states_min
				}
			}
			max = com_movement_religious_minority_secession_target_fraction_of_states_max #vanilla value: 0.75
		}
	}

	additional_radicalism_factors = {
		add = {
			if = {
				limit = { Grey_PLP_is_active = yes } 
				value = "owner.average_sol_for_religion(owner.religion)"
			}
			else = { # Vanilla
				value = "owner.average_sol"
			}
			subtract = "owner.average_sol_for_religion(scope:religion)"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_RELIGION"
		}

        subtract = {
            value = 0.5
            subtract = political_movement_identity_support
            min = 0
            desc = "LOW_MOVEMENT_IDENTITY_SUPPORT"
        }

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								religion ?= ROOT.religion
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_RELIGIOUS_REVOLUTION"
				value = 0.15
			}
		}

		if = {
			limit = {
				owner = {
					any_political_movement = {
						NOT = { this = root } # Insurrectionary movements should not make themselves more insurrectionary
						is_insurrectionary = yes
					}
				}
			}
			add = {
				desc = "INSURRECTIONARY_MOVEMENTS"
				owner = {
					every_political_movement = {
						limit = {
							NOT = { this = root } # Insurrectionary movements should not make themselves more insurrectionary
							is_insurrectionary = yes
						}
						if = {
							limit = {
								exists = this.political_movement_identity_support
							}
							add = { # Identity support can get a bit high
								value = this.political_movement_identity_support
								multiply = 0.5
							}
						}
						else = {
							add = this.political_movement_support
						}
					}
				}
				min = 0.05
				max = 0.25
			}
		}
    }
}
